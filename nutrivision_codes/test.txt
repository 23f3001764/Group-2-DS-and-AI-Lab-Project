## Step 6 — Using the app

1. **Upload** a photo of food with a ₹10 coin flat in frame, **or** use the **camera tab**.
2. Click **🔍 Analyse**.
3. Results appear as:
   - **PCA crop gallery** — each detected food with ellipse axes drawn.
   - **Nutrition table** — food class, estimated weight, and all nutrient values.
   - **LLM Reasoning** (expandable) — geometry model and density explanation.

### If no coin is detected

The app will stop early and show:
> 🪙 **No coin detected in the image. Please include a ₹10 RBI coin (flat, unobstructed)...**

Add the coin to the photo and retry.

### Example images

Drop `.jpg` / `.png` files into `test_images/` and restart the app —
they will appear as clickable examples in the UI.

## Troubleshooting

| Issue | Fix |
|-------|-----|
| `SAM3 not found` | Check `HF_TOKEN` is set and the model is accessible at `facebook/sam3` |
| `FileNotFoundError: best_convnextv2_tiny.pt` | Set `CKPT_PATH` env var to correct path |
| `OLLAMA_API_KEY` errors | Set the env var; check your Ollama Cloud account |
| App runs but GPU OOM | Reduce batch size or use a larger GPU instance |
| Gradio port not accessible | Open port 7860 in the Lightning AI Ports tab |

---

## Key design decisions

- **No coin → hard stop:** The pipeline raises `CoinNotFoundError` and the Gradio
  UI shows a clear user-facing message. No fallback scale is used.
- **Rounded dimensions:** `width_cm` and `height_cm` are rounded to the nearest
  integer before being embedded in the PCA image labels.
- **LLM gets only the image:** The prompt no longer passes `width_cm`/`height_cm`
  as text. The LLM reads the `W=X cm` / `H=X cm` labels drawn on the PCA image
  (PC1 = red arrow, PC2 = green arrow).

```python
# Session helpers
# ════════════════════════════════════════════════════════════════════════════

def _new_session_dir() -> str:
    return tempfile.mkdtemp(prefix="food_nutr_")


def _cleanup_session(session_dir: str | None) -> None:
    if session_dir and os.path.isdir(session_dir):
        shutil.rmtree(session_dir, ignore_errors=True)


def _prune_old_temp_dirs() -> None:
    temp_base = tempfile.gettempdir()
    while True:
        now = time.time()
        try:
            for name in os.listdir(temp_base):
                if not name.startswith("food_nutr_"):
                    continue
                path = os.path.join(temp_base, name)
                if os.path.isdir(path) and (now - os.path.getmtime(path)) > 3600:
                    shutil.rmtree(path, ignore_errors=True)
        except Exception:
            pass
        time.sleep(3600)


threading.Thread(target=_prune_old_temp_dirs, daemon=True).start()

# Progress markdown builder
# ════════════════════════════════════════════════════════════════════════════

def _progress_md(current_step: int, done: bool = False,
                 error: bool = False, paused: bool = False) -> str:
    lines = []
    for i, label in enumerate(_STEP_LABELS, start=1):
        if done or i < current_step:
            lines.append(f"✅  ~~{label}~~")
        elif i == current_step:
            if error:
                lines.append(f"❌  **{label}**")
            elif paused:
                lines.append(f"⏸️  **{label}** — *awaiting your review*")
            else:
                lines.append(f"⏳  **{label}** …")
        else:
            lines.append(f"⬜  {label}")
    return "\n\n".join(lines)

# Nutrition table builder
# ════════════════════════════════════════════════════════════════════════════

def _build_table(results: list) -> pd.DataFrame:
    rows = []
    for r in results:
        nut = r.get("nutrition") or {}
        src = r.get("nutrition_source", "json")
        src_icon = "🌐" if "web" in src else "📋"
        row = {
            "Food":       f"{r['food_class']} {src_icon}",
            "Weight (g)": f"{r['weight_g']:.0f}",
        }
        for col, label in zip(NUTRIENT_COLS, NUTRIENT_LABELS):
            val = nut.get(col)
            row[label] = f"{val:.1f}" if val is not None else "—"
        rows.append(row)
    return pd.DataFrame(rows)

def _crop_path_to_pil(crop_path: str) -> Image.Image | None:
    if crop_path and os.path.exists(crop_path):
        return Image.open(crop_path)
    return None


# ════════════════════════════════════════════════════════════════════════════
# Phase 1 callback  — runs on "Analyse" click
# ════════════════════════════════════════════════════════════════════════════

def phase1_analyse(image, session_dir: str | None):
    """
    Generator. Runs SAM3 → Coin → ConvNeXtV2, then pauses.

    Yields a flat tuple with all outputs in the order declared in run_btn.click outputs.
    """
    _cleanup_session(session_dir)
    new_dir = _new_session_dir()

    # Build the base "empty" update tuple.
    # Outputs (in order):
    #   progress_out, annotated_out, run_btn,
    #   review_group (visible),
    #   seg_row_0..MAX_SEGMENTS-1 (visible),
    #   label_0..MAX_SEGMENTS-1 (value),
    #   ignore_0..MAX_SEGMENTS-1 (value),
    #   crop_0..MAX_SEGMENTS-1 (value),
    #   seg_header_0..MAX_SEGMENTS-1 (value),
    #   table_out, reasoning_out, intermediate_state, session_dir_state

    n = MAX_SEGMENTS

    def _base_yield(progress, run_btn_update, review_visible=False, seg_visible=None):
        seg_vis = seg_visible or [False] * n
        return (
            progress,                                           # progress_out
            None,                                               # annotated_out
            run_btn_update,                                     # run_btn
            gr.update(visible=review_visible),                  # review_group
            *[gr.update(visible=seg_vis[i]) for i in range(n)],# seg_row_0..n-1
            *["" for _ in range(n)],                           # label_0..n-1
            *[False for _ in range(n)],                        # ignore_0..n-1
            *[None for _ in range(n)],                         # crop_0..n-1
            *["" for _ in range(n)],                           # seg_header_0..n-1
            pd.DataFrame(),                                     # table_out
            "",                                                 # reasoning_out
            None,                                               # intermediate_state
            new_dir,                                            # session_dir_state
        )

    if image is None:
        yield _base_yield(
            "⚠️ **No image provided.** Upload a photo or use the camera.",
            gr.update(interactive=True, value="🔍 Analyse"),
        )
        return

    pil_img = Image.fromarray(image)

    try:
        for step_idx, step_label, payload in run_until_review(pil_img, new_dir):

            if payload is None:
                yield _base_yield(
                    _progress_md(step_idx),
                    gr.update(interactive=False, value="⏳ Analysing…"),
                )

            elif isinstance(payload, dict) and "error" in payload:
                yield _base_yield(
                    _progress_md(step_idx, error=True) + f"\n\n⚠️ **{payload['error']}**",
                    gr.update(interactive=True, value="🔍 Analyse"),
                )
                return

            else:
                # Phase 1 complete — build review panel values
                segments = payload["segments"]
                n_segs   = min(len(segments), n)

                seg_vis     = [i < n_segs for i in range(n)]
                labels      = [""] * n
                ignores     = [False] * n
                crops       = [None] * n
                headers     = [""] * n

                for i, seg in enumerate(segments[:n]):
                    labels[i]  = seg["food_class"] or ""
                    ignores[i] = False
                    crops[i]   = _crop_path_to_pil(seg.get("crop_path", ""))
                    conf_str   = f"  (conf {seg['cls_conf']:.0%})" if seg["food_class"] else ""
                    badge      = " — ⚠️ unrecognised" if not seg["food_class"] else ""
                    headers[i] = f"**Segment {i + 1}{badge}**{conf_str}"

                n_classified = sum(1 for s in segments if s["food_class"])
                summary = (
                    f"\n\n**{len(segments)} segment{'s' if len(segments) > 1 else ''} detected** "
                    f"({n_classified} classified, "
                    f"{len(segments) - n_classified} unrecognised) — "
                    f"review and confirm below."
                )

                yield (
                    _progress_md(REVIEW_STEP, paused=True) + summary,  # progress_out
                    None,                                                # annotated_out
                    gr.update(interactive=True, value="🔍 Analyse"),    # run_btn
                    gr.update(visible=True),                             # review_group
                    *[gr.update(visible=seg_vis[i]) for i in range(n)], # seg_row_0..n-1
                    *labels,                                             # label_0..n-1
                    *ignores,                                            # ignore_0..n-1
                    *crops,                                              # crop_0..n-1
                    *headers,                                            # seg_header_0..n-1
                    pd.DataFrame(),                                      # table_out
                    "",                                                  # reasoning_out
                    payload,                                             # intermediate_state
                    new_dir,                                             # session_dir_state
                )
                return

    except Exception as exc:
        yield _base_yield(
            f"❌ **Pipeline error (phase 1):** {exc}",
            gr.update(interactive=True, value="🔍 Analyse"),
        )


# ════════════════════════════════════════════════════════════════════════════
# Phase 2 callback  — triggered by "Confirm & Continue" button
# ════════════════════════════════════════════════════════════════════════════

def phase2_confirm(intermediate, session_dir, *label_and_ignore_values):
    """
    Generator. Receives user labels + ignore flags, runs PCA → LLM → Nutrition.

    label_and_ignore_values: flat list of MAX_SEGMENTS label strings followed by
                             MAX_SEGMENTS ignore booleans (same order as Gradio inputs).

    Yields:
        (progress_md, table_df, reasoning_md, annotated_img,
         review_group_update, run_btn_update)
    """
    n = MAX_SEGMENTS
    label_values  = label_and_ignore_values[:n]
    ignore_values = label_and_ignore_values[n:]

    def _err(msg):
        return (
            f"❌ **{msg}**",
            pd.DataFrame(),
            "",
            None,
            gr.update(visible=True),
            gr.update(interactive=True, value="🔍 Analyse"),
        )

    if intermediate is None:
        yield _err("No analysis to confirm. Please run Analyse first.")
        return

    segments = intermediate["segments"]

    # Build ordered user_labels list matching segments order
    user_labels = []
    for i, seg in enumerate(segments):
        if i >= n:
            user_labels.append(None)
            continue
        ignored = ignore_values[i] if i < len(ignore_values) else False
        label   = (label_values[i] or "").strip() if i < len(label_values) else ""
        if ignored or not label:
            user_labels.append(None)
        else:
            user_labels.append(label.lower())

    save_dir = session_dir or _new_session_dir()
    accumulated_results = []

    try:
        for step_idx, step_label, payload in run_from_review(
                user_labels, intermediate, save_dir):

            if payload is None:
                yield (
                    _progress_md(step_idx),
                    _build_table(accumulated_results),
                    "",
                    None,
                    gr.update(visible=False),
                    gr.update(interactive=False, value="⏳ Processing…"),
                )

            elif isinstance(payload, tuple):
                kind, data = payload

                if kind == "row":
                    accumulated_results.append(data)
                    reasoning_md = _build_reasoning(accumulated_results)
                    yield (
                        _progress_md(step_idx),
                        _build_table(accumulated_results),
                        reasoning_md,
                        None,
                        gr.update(visible=False),
                        gr.update(interactive=False, value="⏳ Processing…"),
                    )

                elif kind == "done":
                    annotated_path = data
                    annotated_img  = (
                        annotated_path
                        if annotated_path and os.path.exists(annotated_path)
                        else None
                    )
                    n_items    = len(accumulated_results)
                    total_kcal = sum(
                        (r.get("nutrition") or {}).get("calories_kcal", 0) or 0
                        for r in accumulated_results
                    )
                    kcal_str = f" · **{total_kcal:.0f} kcal** total" if total_kcal else ""
                    summary  = f"\n\n**{n_items} item{'s' if n_items > 1 else ''} processed**{kcal_str}"

                    yield (
                        _progress_md(N_STEPS, done=True) + summary,
                        _build_table(accumulated_results),
                        _build_reasoning(accumulated_results),
                        annotated_img,
                        gr.update(visible=False),
                        gr.update(interactive=True, value="🔍 Analyse"),
                    )

                elif kind == "error":
                    yield _err(data)
                    return

    except Exception as exc:
        yield (
            f"❌ **Pipeline error (phase 2):** {exc}",
            _build_table(accumulated_results),
            "",
            None,
            gr.update(visible=True),
            gr.update(interactive=True, value="🔍 Analyse"),
        )


def _build_reasoning(results: list) -> str:
    parts = []
    for r in results:
        density_src  = r.get("density_source", "—")
        density_icon = "📋" if "food_density" in density_src else "🧠"
        src_label    = r.get("nutrition_source", "json")
        src_tag      = "🌐 web" if "web" in src_label else "📁 json"
        parts.append(
            f"### {r['food_class']}  _(conf {r['cls_conf']:.0%})_\n"
            f"- **Weight:** {r['weight_g']:.0f} g\n"
            f"- **Geometry:** {r['geometry']}\n"
            f"- **Density:** {r['density']:.3f} g/cm³  "
            f"{density_icon} _{density_src}_\n"
            f"- **Nutrition source:** {src_tag}\n"
            f"- **Reasoning:** {r['reasoning']}\n"
        )
    return "\n---\n".join(parts)
```

```python
"""
models.py  —  Lazy loading of SAM3 and ConvNeXtV2.

Both models are loaded once on the first call and cached globally.

Fix applied:
  [FIX 15] Removed unused import of CONVNEXT_CONF_THRESH — it is only used
           in pipeline.py (passed directly to classify_crop). Importing it
           here caused a misleading "imported but unused" warning.
"""

import torch
import timm
import albumentations as A
from albumentations.pytorch import ToTensorV2
from transformers import Sam3Processor, Sam3Model

from config import CKPT_PATH

# ── Globals (populated on first use) ──────────────────────────────────────
_device           = None
_sam_model        = None
_sam_processor    = None
_cls_model        = None
_idx_to_class     = None
_inference_tf     = None


def get_device() -> str:
    global _device
    if _device is None:
        _device = "cuda" if torch.cuda.is_available() else "cpu"
    return _device


def get_sam():
    """Return (sam_model, sam_processor), loading on first call."""
    global _sam_model, _sam_processor
    if _sam_model is None:
        device = get_device()
        print(f"[models] Loading SAM3 on {device} ...")
        _sam_model     = Sam3Model.from_pretrained("facebook/sam3").to(device)
        _sam_processor = Sam3Processor.from_pretrained("facebook/sam3")
        print("[models] SAM3 ready.")
    return _sam_model, _sam_processor


def get_classifier():
    """Return (cls_model, idx_to_class, inference_transform), loading on first call."""
    global _cls_model, _idx_to_class, _inference_tf
    if _cls_model is None:
        device = get_device()
        print(f"[models] Loading ConvNeXtV2 from '{CKPT_PATH}' ...")
        ckpt = torch.load(CKPT_PATH, map_location=device, weights_only=False)

        class_to_idx  = ckpt["class_to_idx"]
        _idx_to_class = {v: k for k, v in class_to_idx.items()}
        num_classes   = ckpt["cfg"]["num_classes"]
        img_size      = ckpt["cfg"]["img_size"]
        model_name    = ckpt["cfg"]["model_name"]

        _cls_model = timm.create_model(
            model_name, pretrained=False, num_classes=num_classes, drop_rate=0.0
        )
        _cls_model.load_state_dict(ckpt["model_state"])
        _cls_model = _cls_model.to(device, memory_format=torch.channels_last)
        _cls_model.eval()

        MEAN = [0.485, 0.456, 0.406]
        STD  = [0.229, 0.224, 0.225]
        _inference_tf = A.Compose([
            A.Resize(height=int(img_size * 1.14), width=int(img_size * 1.14)),
            A.CenterCrop(height=img_size, width=img_size),
            A.Normalize(mean=MEAN, std=STD),
            ToTensorV2(),
        ])

        print(f"[models] ConvNeXtV2 ready — {num_classes} classes: "
              f"{list(class_to_idx.keys())}")

    return _cls_model, _idx_to_class, _inference_tf
```

# ── Core fetch ─────────────────────────────────────────────────────────────

def fetch_nutrition_from_web(food_name: str) -> dict:
    """
    Fetch nutrition per 100g for food_name via TavilySearch + LLM.
    Returns a dict with keys matching NUTRIENT_COLS.
    """
    food_name = food_name.strip().lower()
    base = {col: None for col in NUTRIENT_COLS}
    base["food_class"]       = food_name
    base["nutrition_source"] = "web_failed"

    cleaned = re.sub(r"[^a-z0-9 ]", "", food_name).strip()
    if len(cleaned) < 2:
        print(f"  [nutrition_fetch] Skipping '{food_name}' — too short after sanitise")
        return base

    # ── 1. Tavily search ────────────────────────────────────────────────
    try:
        tool    = _get_tavily()
        query   = f"{cleaned} nutrition facts per 100g calories protein carbs fat"
        # TavilySearch.invoke() returns a list of result dicts
        results = tool.invoke({"query": query})

        snippets = []
        for r in results:
            # Each result has keys: url, content, score, title
            content = r.get("content", "").strip()
            if content:
                snippets.append(content[:600])
        search_context = "\n\n".join(snippets[:5])

    except Exception as exc:
        print(f"  [nutrition_fetch] Tavily failed for '{food_name}': {exc}")
        return base

    if not search_context:
        print(f"  [nutrition_fetch] No Tavily content for '{food_name}'")
        return base

    # ── 2. LLM parse ────────────────────────────────────────────────────
    prompt = f"""You are a nutrition database assistant.

Food item: {food_name}

Web search results:
{search_context}

Extract the nutritional values per 100g from the search results above.
Return ONLY a valid JSON object with exactly these keys (use null if unknown):

{{
  "calories_kcal": <number or null>,
  "carbohydrates_g": <number or null>,
  "protein_g": <number or null>,
  "fats_g": <number or null>,
  "free_sugar_g": <number or null>,
  "fibre_g": <number or null>,
  "sodium_mg": <number or null>,
  "calcium_mg": <number or null>,
  "iron_mg": <number or null>,
  "vitamin_c_mg": <number or null>,
  "folate_ug": <number or null>
}}

Do NOT include any explanation, markdown, or extra text — only the JSON object.
"""

    try:
        raw  = _nutrition_llm.invoke([HumanMessage(content=prompt)])
        text = raw.content if hasattr(raw, "content") else str(raw)
        text = re.sub(r"```(?:json)?", "", text).strip().strip("`").strip()

        parsed = NutritionPer100g(**json.loads(text))
        for col in NUTRIENT_COLS:
            base[col] = getattr(parsed, col, None)

        filled = sum(1 for col in NUTRIENT_COLS if base[col] is not None)
        base["nutrition_source"] = "web" if filled >= 4 else "web_partial"
        print(f"  [nutrition_fetch] '{food_name}' → {filled}/{len(NUTRIENT_COLS)} fields ({base['nutrition_source']})")

    except Exception as exc:
        print(f"  [nutrition_fetch] LLM parse failed for '{food_name}': {exc}")
        base["nutrition_source"] = "web_failed"

    return base


def scale_web_nutrition(nutrition_per_100g: dict, weight_g: float) -> dict:
    """Scale per-100g nutrition to actual weight."""
    factor = weight_g / 100.0
    scaled = {}
    for col in NUTRIENT_COLS:
        raw = nutrition_per_100g.get(col)
        scaled[col] = round(raw * factor, 2) if raw is not None else None
    scaled["food_class"]       = nutrition_per_100g.get("food_class", "")
    scaled["nutrition_source"] = nutrition_per_100g.get("nutrition_source", "web")
    return scaled


# ── Batch helper (parallel) ────────────────────────────────────────────────

def fetch_nutrition_batch(food_names: list[str],
                          max_workers: int = 4) -> dict[str, dict]:
    """Fetch nutrition for multiple unknown foods in parallel."""
    results: dict[str, dict] = {}
    with ThreadPoolExecutor(max_workers=max_workers) as pool:
        future_to_name = {
            pool.submit(fetch_nutrition_from_web, name): name
            for name in food_names
        }
        for future in as_completed(future_to_name):
            name = future_to_name[future]
            try:
                results[name] = future.result()
            except Exception as exc:
                print(f"  [nutrition_fetch] batch error for '{name}': {exc}")
                results[name] = {col: None for col in NUTRIENT_COLS}
                results[name]["food_class"] = name
                results[name]["nutrition_source"] = "web_failed"
    return results

```python
"""

class Detections:

# Image pre-processing  (quality reduction before pipeline)
# ════════════════════════════════════════════════════════════════════════════

def _preprocess_image(image_pil: Image.Image) -> Image.Image:
    """
    Resize to PREPROCESS_MAX_DIM and JPEG-compress before anything else.

    This reduces memory pressure and speeds up SAM3 without materially
    affecting segmentation quality — the coin and food regions are still
    clearly visible at the output resolution.
    """
    image_pil = image_pil.convert("RGB")
    w, h = image_pil.size
    max_dim = max(w, h)

    if max_dim > PREPROCESS_MAX_DIM:
        scale = PREPROCESS_MAX_DIM / max_dim
        new_w, new_h = int(w * scale), int(h * scale)
        print(f"  [preprocess] resize {w}×{h} → {new_w}×{new_h} "
              f"(scale {scale:.3f})")
        image_pil = image_pil.resize((new_w, new_h), Image.LANCZOS)
    else:
        print(f"  [preprocess] no resize needed ({w}×{h})")

    buf = io.BytesIO()
    image_pil.save(buf, format="JPEG", quality=PREPROCESS_JPEG_QUALITY,
                   optimize=True)
    buf.seek(0)
    result = Image.open(buf).copy()   # .copy() detaches from the BytesIO buffer
    print(f"  [preprocess] JPEG quality={PREPROCESS_JPEG_QUALITY}  "
          f"compressed size={buf.tell() / 1024:.0f} KB")
    return result

def _results_to_dets(results) -> Detections:
    if len(results["scores"]) == 0:
        return _empty_dets()
    masks  = np.stack([m.cpu().numpy().astype(bool) for m in results["masks"]], axis=0)
    boxes  = results["boxes"].cpu().numpy().astype(np.float32)
    scores = results["scores"].cpu().numpy().astype(np.float32)
    return Detections(boxes, scores, masks)


def get_dets(image: Image.Image, prompt: str,
             conf_thresh: float = CONF_THRESH) -> Detections:
    sam_model, sam_processor = get_sam()
    device = get_device()
    inputs = sam_processor(
        images=image, text=prompt, return_tensors="pt"
    ).to(device)
    with torch.no_grad():
        outputs = sam_model(**inputs)
    results = sam_processor.post_process_instance_segmentation(
        outputs,
        threshold=0.0,
        mask_threshold=0.5,
        target_sizes=inputs["original_sizes"].tolist()
    )[0]
    dets = _results_to_dets(results)
    if len(dets) == 0:
        return dets
    return dets.filter(dets.confidence >= conf_thresh)


def _mask_iou(m1, m2):
    inter = np.logical_and(m1, m2).sum()
    union = np.logical_or(m1,  m2).sum()
    return float(inter / union) if union > 0 else 0.0


def _mask_containment(m_small, m_large):
    inter = np.logical_and(m_small, m_large).sum()
    area  = m_small.sum()
    return float(inter / area) if area > 0 else 0.0


def merge_detections(det_p1, det_p2, det_p3, prompt_labels,
                     iou_threshold=0.30, containment_threshold=0.25):
    masks, xyxys, confs, tags = [], [], [], []
    SPECIFIC_TAGS  = {0, 1}
    FOOD_ONLY_TAGS = {2}

    for src_idx, dets in enumerate([det_p1, det_p2, det_p3]):
        if dets is None or len(dets) == 0:
            continue
        for i in range(len(dets)):
            masks.append(dets.mask[i])
            xyxys.append(dets.xyxy[i])
            confs.append(float(dets.confidence[i]))
            tags.append(src_idx)

    n = len(masks)
    if n == 0:
        return _empty_dets(), []

    iou_mat = np.zeros((n, n), dtype=np.float32)
    for i in range(n):
        for j in range(i + 1, n):
            v = _mask_iou(masks[i], masks[j])
            iou_mat[i, j] = iou_mat[j, i] = v

    visited = [False] * n
    groups  = []
    for start in range(n):
        if visited[start]:
            continue
        group, queue = [], [start]
        visited[start] = True
        while queue:
            cur = queue.pop()
            group.append(cur)
            for nb in range(n):
                if not visited[nb] and iou_mat[cur, nb] >= iou_threshold:
                    visited[nb] = True
                    queue.append(nb)
        groups.append(group)

    keep_indices = []
    for group in groups:
        containers = [idx for idx in group if tags[idx] in SPECIFIC_TAGS]
        food_only  = [idx for idx in group if tags[idx] in FOOD_ONLY_TAGS]
        if containers:
            keep_indices.append(max(containers, key=lambda idx: confs[idx]))
        elif food_only:
            keep_indices.append(max(food_only, key=lambda idx: confs[idx]))

    container_kept = [masks[i] for i in keep_indices if tags[i] in SPECIFIC_TAGS]
    final_indices  = []
    for i in keep_indices:
        if tags[i] in FOOD_ONLY_TAGS:
            if any(_mask_containment(masks[i], c) >= containment_threshold
                   for c in container_kept):
                continue
        final_indices.append(i)

    if not final_indices:
        return _empty_dets(), []

    tag_name = {i: lbl for i, lbl in enumerate(prompt_labels)}
    return Detections(
        xyxy       = np.array([xyxys[i] for i in final_indices]),
        confidence = np.array([confs[i]  for i in final_indices]),
        mask       = np.array([masks[i]  for i in final_indices]),
    ), [tag_name.get(tags[i], f"prompt_{tags[i]}") for i in final_indices]


# ════════════════════════════════════════════════════════════════════════════
# PCA geometry
# ════════════════════════════════════════════════════════════════════════════

def pca_geometry(mask: np.ndarray):
    ys, xs   = np.where(mask)
    coords   = np.stack([xs, ys], axis=1).astype(np.float32)
    center   = coords.mean(axis=0)
    pca      = skPCA(n_components=2).fit(coords)
    stds     = np.sqrt(pca.explained_variance_)
    width_px  = 4.0 * stds[0]
    height_px = 4.0 * stds[1]
    pc1       = pca.components_[0]
    angle_deg = np.degrees(np.arctan2(pc1[1], pc1[0]))
    return center, (width_px, height_px), angle_deg, pca.components_


# ════════════════════════════════════════════════════════════════════════════
# Coin scale extraction
# ════════════════════════════════════════════════════════════════════════════

def extract_coin_scale(coin_dets: Detections, img_np: np.ndarray,
                       save_dir: str) -> float:
    if coin_dets is None or len(coin_dets) == 0:
        raise CoinNotFoundError(
            "No coin detected in the image.\n\n"
            "Please include a ₹10 RBI coin (flat, unobstructed) in the photo "
            "and try again."
        )

    best_idx  = int(np.argmax(coin_dets.confidence))
    coin_mask = coin_dets.mask[best_idx]
    coin_xyxy = coin_dets.xyxy[best_idx]

    if coin_mask.sum() < 50:
        raise CoinNotFoundError(
            "Coin mask is too small — coin may be partially out of frame or occluded."
        )

    _, (w_px, h_px), _, _ = pca_geometry(coin_mask)
    major_px  = w_px
    roundness = h_px / w_px if w_px > 0 else 0

    if roundness < 0.6:
        print(f"  WARNING: coin may be tilted (roundness={roundness:.2f}).")

    px_per_cm = major_px / COIN_DIAMETER_CM
    print(f"  Coin  major={major_px:.1f}px  minor={h_px:.1f}px  "
          f"roundness={roundness:.2f}  =>  {px_per_cm:.3f} px/cm")

    x1, y1, x2, y2 = map(int, coin_xyxy)
    c_img  = img_np[y1:y2, x1:x2]
    c_mask = coin_mask[y1:y2, x1:x2]
    c_ctr, (cm_w, cm_h), c_ang, _ = pca_geometry(c_mask)

    fig = Figure(figsize=(4, 4))
    FigureCanvasAgg(fig)
    ax = fig.add_subplot(111)
    ax.imshow(c_img)
    ov = np.zeros((*c_mask.shape, 4), dtype=np.float32)
    ov[c_mask] = [1.0, 0.85, 0.0, 0.4]
    ax.imshow(ov)
    ax.add_patch(Ellipse(
        xy=c_ctr, width=cm_w, height=cm_h, angle=c_ang,
        edgecolor="#ffcc00", facecolor="none", linewidth=2.5
    ))
    ax.plot(*c_ctr, "o", color="white", markersize=7,
            markeredgecolor="#333", markeredgewidth=1.5)
    ax.set_title(f"Coin: {major_px:.0f}px = {COIN_DIAMETER_CM}cm  "
                 f"=> {px_per_cm:.2f} px/cm", fontsize=9)
    ax.axis("off")
    fig.tight_layout()
    fig.savefig(os.path.join(save_dir, "coin_debug.png"), dpi=120, bbox_inches="tight")

    return px_per_cm


# ════════════════════════════════════════════════════════════════════════════
# ConvNeXtV2 classification
# ════════════════════════════════════════════════════════════════════════════

def classify_crop(crop_rgb: np.ndarray, conf_thresh: float = CONVNEXT_CONF_THRESH):
    cls_model, idx_to_class, inference_tf = get_classifier()
    device = get_device()
    tensor = inference_tf(image=crop_rgb)["image"].unsqueeze(0)
    tensor = tensor.to(device, memory_format=torch.channels_last)
    autocast_device = "cuda" if device == "cuda" else "cpu"
    with torch.no_grad(), torch.amp.autocast(autocast_device):
        probs = torch.softmax(cls_model(tensor), dim=1)[0]
    top_prob, top_idx = probs.max(0)
    conf = top_prob.item()
    if conf < conf_thresh:
        return None, conf
    return idx_to_class[top_idx.item()], conf

# Classify + crop helper  (Phase 1)

def _classify_and_crop(img_np, food_dets, food_labels, save_dir):
    """
    Run ConvNeXtV2 on every food detection.
    Returns list of segment dicts — ALL segments, classified or not.
    Also saves a crop PNG for every segment for the review panel.
    """
    segments = []
    for i, (mask, xyxy, label) in enumerate(
            zip(food_dets.mask, food_dets.xyxy, food_labels)):
        H_img, W_img = img_np.shape[:2]
        x1 = max(0, int(xyxy[0]))
        y1 = max(0, int(xyxy[1]))
        x2 = min(W_img, int(xyxy[2]))
        y2 = min(H_img, int(xyxy[3]))
        crop_mask     = mask[y1:y2, x1:x2]
        original_crop = img_np[y1:y2, x1:x2].copy()
        masked_crop   = original_crop.copy()
        masked_crop[~crop_mask] = 128

        food_class, conf = classify_crop(masked_crop)

        # Save crop regardless of whether classification succeeded
        crop_path = save_crop_image(
            original_crop, crop_mask, i,
            food_class if food_class else "unlabeled",
            save_dir,
        )

        segments.append({
            "seg_idx":    i,
            "sam_label":  label,
            "food_class": food_class,       # None if below threshold
            "cls_conf":   conf,
            "mask":       mask,             # full-image mask
            "xyxy":       xyxy,
            "crop_img":   original_crop,
            "crop_mask":  crop_mask,
            "crop_path":  crop_path,        # path for review panel
        })
    return segments

def _save_pca_image_for_llm(item: dict, pixels_per_cm: float,
                             seg_idx: int, save_dir: str) -> dict | None:
    crop_mask = item["crop_mask"]
    crop_img  = item["crop_img"]

    if crop_mask.sum() < 50:

    center, (width_px, height_px), angle_deg, comps = pca_geometry(crop_mask)

    width_cm  = round(width_px  / pixels_per_cm, 1)

    print(f"  {item['food_class']:<24}  W={width_cm} cm  H={height_cm} cm")

    fig = Figure(figsize=(6, 6))
    FigureCanvasAgg(fig)
    ax = fig.add_subplot(111)

    ov = np.zeros((*crop_mask.shape, 4), dtype=np.float32)
    ov[crop_mask] = [0.2, 0.6, 1.0, 0.35]
    ax.imshow(ov)

    cx, cy = center
    ax.add_patch(Ellipse(
        xy=(cx, cy), width=width_px, height=height_px, angle=angle_deg,
        edgecolor="#00d4ff", facecolor="none", linewidth=2, linestyle="--"
    ))

    pc1 = comps[0]; hw = width_px / 2
    ax.annotate("",

    ax.text(cx + pc1[0]*hw + 4, cy + pc1[1]*hw - 8, f"W={width_cm} cm",
            color="#cc0000", fontsize=9, fontweight="bold",
            bbox=dict(facecolor="white", alpha=0.75, pad=2, edgecolor="none"))

    pc2 = comps[1]; hh = height_px / 2

# LLM weight estimation
# ════════════════════════════════════════════════════════════════════════════

class WeightEstimate(BaseModel):
    weight_g:          float = Field(description="Single best-estimate weight in grams.")
    geometry:          str   = Field(description="Geometric shape assumed.")
    density_g_per_cm3: float = Field(description="Density value — copy verbatim from prompt.")
    reasoning:         str   = Field(description="One sentence: shape, volume formula used.")
    density_source:    str   = Field(default="", description="Populated by pipeline, not LLM.")


_parser = PydanticOutputParser(pydantic_object=WeightEstimate)

_llm = ChatOpenAI(
    api_key=OLLAMA_API_KEY,
    base_url=OLLAMA_BASE_URL,
    model=OLLAMA_MODEL,
    temperature=0,
    seed=42,
)


def _img_b64(path: str) -> str:
    with open(path, "rb") as f:
        return base64.b64encode(f.read()).decode("utf-8")


def estimate_weight(item: dict) -> WeightEstimate:
    density_val, density_src = _get_density(item["food_class"])
    b64 = _img_b64(item["pca_img_path"])

    prompt = (
        f"You are a food portion weight estimation expert.\n\n"
        f"Food item: {item['food_class']}\n\n"
        f"Step 1 — Measured dimensions (from PCA + calibrated coin scale):\n"
        f"  • Major axis width  (W): {item['width_cm']} cm\n"
        f"  • Minor axis height (H): {item['height_cm']} cm\n\n"
        f"Step 2 — Visual reference:\n"
        f"The attached image shows the segmented food item with its PCA ellipse overlay. "
        f"Use the numeric dimensions above as authoritative measurements.\n\n"
        f"Step 3 — Estimate the weight of this food portion.\n"
        f"Choose a realistic geometric model, compute volume, multiply by density.\n\n"
        f"{_parser.get_format_instructions()}"
    )

    msg = HumanMessage(content=[
        {"type": "image_url", "image_url": {"url": f"data:image/png;base64,{b64}"}},
        {"type": "text",      "text": prompt},
    ])

    try:
        raw_response = _llm.invoke([msg])
        result = _parser.invoke(raw_response)
    except (OutputParserException, Exception) as exc:
        print(f"  [estimate_weight] parse failed for '{item['food_class']}': {exc}")
        result = WeightEstimate(
            weight_g=100.0,
            geometry="unknown (parse error)",
            density_g_per_cm3=density_val,
            reasoning=f"Defaulted to 100 g — LLM parse error: {str(exc)[:120]}",
            density_source=density_src,
        )
        return result

    result.density_g_per_cm3 = density_val
    result.density_source = density_src
    return result


# ════════════════════════════════════════════════════════════════════════════
# Nutrition lookup (JSON first, web fallback)
# ════════════════════════════════════════════════════════════════════════════

def _fuzzy_lookup(food_class: str) -> dict | None:
    key  = food_class.strip().lower()
    if key in NUTRITION_DB:
        return NUTRITION_DB[key]
    norm = re.sub(r"[^a-z0-9 ]", "", key)
    for db_key, entry in NUTRITION_DB.items():
        db_norm = re.sub(r"[^a-z0-9 ]", "", db_key)
        if db_norm in norm or norm in db_norm:
            return entry
    return None


def scale_nutrition_json(food_class: str, weight_g: float) -> dict | None:
    """Returns scaled nutrition from local JSON, or None if not found."""
    entry = _fuzzy_lookup(food_class)
    if entry is None:
        return None
    factor = weight_g / 100.0
    scaled = {"food_class": food_class, "weight_g": round(weight_g, 1),
              "nutrition_source": "json"}
    for col in NUTRIENT_COLS:
        raw_val = entry.get(col)
        scaled[col] = round(raw_val * factor, 2) if raw_val is not None else None
    return scaled


# ════════════════════════════════════════════════════════════════════════════
# Pipeline step counts
# ════════════════════════════════════════════════════════════════════════════

_STEP_LABELS = [
    "SAM3 Segmentation",         # 1
    "Coin Scale Detection",      # 2
    "Food Classification",       # 3  ← pauses here for review
    "PCA Geometry",              # 4
    "Weight & Nutrition",        # 5
]
N_STEPS = len(_STEP_LABELS)

# Step index at which we pause for user review
REVIEW_STEP = 3


# ════════════════════════════════════════════════════════════════════════════
# Phase 1 — run_until_review()
# ════════════════════════════════════════════════════════════════════════════

def run_until_review(image_pil: Image.Image, save_dir: str):
    """
    Generator — Phase 1 of the pipeline.

    Yields:
      (step_idx, step_label, payload)

    payload is None for progress-only yields.
    On completion yields payload = {
        "segments":      list of segment dicts (ALL segments, classified or not)
        "pixels_per_cm": float
        "sam_image":     PIL Image (resized)
        "img_np":        np.ndarray
    }
    On error yields payload = {"error": str}
    """

    # ── 0. Pre-process (resize + JPEG compress) ───────────────────────────
    image_pil = _preprocess_image(image_pil)

    # ── 1. SAM3 ──────────────────────────────────────────────────────────
    yield (1, _STEP_LABELS[0], None)

    sam_image = _resize_for_sam(image_pil)
    img_np = np.array(sam_image.convert("RGB"))

    det1 = get_dets(sam_image, PROMPT_1)
    if torch.cuda.is_available(): torch.cuda.empty_cache()
    det2 = get_dets(sam_image, PROMPT_2)
    if torch.cuda.is_available(): torch.cuda.empty_cache()
    det3 = get_dets(sam_image, PROMPT_3)
    if torch.cuda.is_available(): torch.cuda.empty_cache()
    coin_dets = get_dets(sam_image, PROMPT_COIN, conf_thresh=COIN_CONF_THRESH)
    if torch.cuda.is_available(): torch.cuda.empty_cache()

    print(f"Food raw → P1:{len(det1)}  P2:{len(det2)}  P3:{len(det3)}")
    print(f"Coin     → {len(coin_dets)}")

    food_dets, food_labels = merge_detections(
        det1, det2, det3,
        prompt_labels=[PROMPT_1, PROMPT_2, PROMPT_3],
        iou_threshold=IOU_THRESH,
        containment_threshold=CONT_THRESH,
    )
    print(f"Food after merge → {len(food_dets)} segments")

    # ── 2. Coin scale ────────────────────────────────────────────────────
    yield (2, _STEP_LABELS[1], None)
    try:
        pixels_per_cm = extract_coin_scale(coin_dets, img_np, save_dir)
    except CoinNotFoundError as e:
        yield (N_STEPS, _STEP_LABELS[-1], {"error": str(e)})
        return

    # ── 3. Classification → save crops → pause ───────────────────────────
    yield (3, _STEP_LABELS[2], None)

    if len(food_dets) == 0:
        yield (REVIEW_STEP, _STEP_LABELS[2], {
            "error": "No food segments detected. Try a clearer photo."
        })
        return

    segments = _classify_and_crop(img_np, food_dets, food_labels, save_dir)
    print(f"Segments for review: {len(segments)} "
          f"({sum(1 for s in segments if s['food_class'])} classified)")

    # Yield the intermediate state — Gradio will pause here for review
    yield (REVIEW_STEP, _STEP_LABELS[2], {
        "segments":      segments,
        "pixels_per_cm": pixels_per_cm,
        "sam_image":     sam_image,
        "img_np":        img_np,
    })


# ════════════════════════════════════════════════════════════════════════════
# Phase 2 — run_from_review()
# ════════════════════════════════════════════════════════════════════════════

def run_from_review(user_labels: list[str | None],
                    intermediate: dict,
                    save_dir: str):
    """
    Generator — Phase 2 of the pipeline.

    user_labels: one entry per segment in intermediate["segments"].
        - str  → use this label (may differ from model prediction)
        - None → user marked as ignore
        "" (empty string) → treated same as None (ignore)

    Yields:
      (step_idx, step_label, partial_result | None)

    partial_result is None for progress-only yields.
    Each item yields:
        ("row", result_dict)          — one completed nutrition row
    Final yield:
        ("done", annotated_path)      — annotated image path (or None)
    On error:
        ("error", message_str)
    """
    segments      = intermediate["segments"]
    pixels_per_cm = intermediate["pixels_per_cm"]
    sam_image     = intermediate["sam_image"]

    # ── Build active items (apply user labels, drop ignored) ─────────────
    active_items = []
    for seg, user_label in zip(segments, user_labels):
        label = (user_label or "").strip()
        if not label:
            continue  # user ignored this segment
        item = dict(seg)       # shallow copy
        item["food_class"] = label.lower()
        active_items.append(item)

    if not active_items:
        yield (N_STEPS, _STEP_LABELS[-1], ("error",
               "No items to process — all segments were ignored."))
        return

    # ── 4. PCA geometry ──────────────────────────────────────────────────
    yield (4, _STEP_LABELS[3], None)

    pca_data = []
    for item in active_items:
        result = _save_pca_image_for_llm(
            item, pixels_per_cm, item["seg_idx"], save_dir)
        if result:
            pca_data.append(result)

    if not pca_data:
        yield (N_STEPS, _STEP_LABELS[-1], ("error",
               "PCA geometry failed for all active items."))
        return

    # ── Identify which foods need web nutrition lookup (not in JSON) ──────
    unknown_foods = [
        item["food_class"] for item in pca_data
        if _fuzzy_lookup(item["food_class"]) is None
    ]
    web_nutrition_cache: dict[str, dict] = {}
    if unknown_foods:
        print(f"  Fetching web nutrition for: {unknown_foods}")
        web_nutrition_cache = fetch_nutrition_batch(unknown_foods)

    # ── 5. LLM weight + nutrition (streaming one row at a time) ──────────
    yield (5, _STEP_LABELS[4], None)

    final_results = []
    mask_by_seg   = {s["seg_idx"]: s["mask"] for s in segments}

    for item in pca_data:
        print(f"\n[{item['seg_idx']}] {item['food_class']} ...")
        est = estimate_weight(item)

        # Nutrition: JSON first, web fallback
        nutrition = scale_nutrition_json(item["food_class"], est.weight_g)
        if nutrition is None:
            per100 = web_nutrition_cache.get(item["food_class"])
            if per100:
                nutrition = scale_web_nutrition(per100, est.weight_g)
                nutrition["weight_g"] = round(est.weight_g, 1)
            else:
                nutrition = {col: None for col in NUTRIENT_COLS}
                nutrition["nutrition_source"] = "web_failed"

        result = {
            "seg_idx":          item["seg_idx"],
            "food_class":       item["food_class"],
            "cls_conf":         item["cls_conf"],
            "width_cm":         item["width_cm"],
            "height_cm":        item["height_cm"],
            "weight_g":         est.weight_g,
            "geometry":         est.geometry,
            "density":          est.density_g_per_cm3,
            "density_source":   est.density_source,
            "reasoning":        est.reasoning,
            "nutrition":        nutrition,
            "nutrition_source": nutrition.get("nutrition_source", "json"),
        }
        final_results.append(result)

        # Stream this row immediately
        yield (5, _STEP_LABELS[4], ("row", result))

    # ── Build combined annotated image ───────────────────────────────────
    full_masks = [
        mask_by_seg[r["seg_idx"]]
        for r in final_results
        if r["seg_idx"] in mask_by_seg
    ]

    annotated_path = None
    if final_results and full_masks:
        annotated_path = save_combined_annotated_image(
            sam_image, final_results, full_masks, save_dir)

    yield (N_STEPS, _STEP_LABELS[-1], ("done", annotated_path))


# ════════════════════════════════════════════════════════════════════════════
# Non-streaming entry point (kept for tests / scripts)
# ════════════════════════════════════════════════════════════════════════════

def run_pipeline(image_pil: Image.Image, save_dir: str):
    """Blocking convenience wrapper — not used by Gradio."""
    intermediate = None
    for _, _, payload in run_until_review(image_pil, save_dir):
        if isinstance(payload, dict) and "error" not in payload:
            intermediate = payload

    if intermediate is None:
        return [], None, "Phase 1 failed"

    # Default: keep all model labels, ignore unclassified
    user_labels = [
        s["food_class"] for s in intermediate["segments"]
    ]

    final_results = []
    annotated_path = None
    for _, _, payload in run_from_review(user_labels, intermediate, save_dir):
        if isinstance(payload, tuple):
            kind, data = payload
            if kind == "row":
                final_results.append(data)
            elif kind == "done":
                annotated_path = data

    return final_results, annotated_path, None
```
